from django.http import HttpResponse

def home(request):
    html_content = """
    <html>
    <style>
    h1{
        color:red;
        text-align: center;
    }
    p{
        color: green;
        text-align: center;
    }
    </style>
    <body>
       <h1>heading</h1>
       <p>this is my first django web app</p>
    
    </body>
    </html>
    """

    return HttpResponse(html_content)

# after we have registere the views to url
from django.urls import path
from myapp.views import home

urlpatterns = [
    path('myapp/', home),
]
